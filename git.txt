git init	  			- гит почва да следи конкретната папка за  промени по нея
git add	 				- добавя файл за примерно пакетиране
git add .				- добавя всички файлове в папката за примерно пакетиране
git commit -m "test coment"		- пакетира add-натите файлове за да можеш да ги качваш 
git push origin master			- прикачваш в интернет 
git clone http:://....			- клонира/копира файловете от подадения url в посочената папка
git pull origin maser			- тегли от интернет
git remote add origin http:://..	- вързваш git конзолата да се синхронизира/следи url-а
